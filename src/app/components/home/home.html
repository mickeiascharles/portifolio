/*
  Historia_Mickeias.c
  Um programa em C que representa a história de vida do usuário (Mickeias)
  com base nas informações que ele compartilhou.

  Como usar:
  gcc Historia_Mickeias.c -o Historia_Mickeias
  ./Historia_Mickeias
*/

#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int day;
    int month;
    int year;
} Date;

typedef struct {
    const char *title;
    const char *description;
    Date date; // data aproximada (quando disponível)
} Event;

typedef struct {
    const char *name;
    Date birthdate;
    const char *birthplace;
    const char *favorites[10];
    int favorites_count;
    const char *idols[10];
    int idols_count;
    Event timeline[32];
    int timeline_count;
} Person;

void printDate(Date d){
    if(d.day==0 && d.month==0 && d.year==0) printf("(data desconhecida)");
    else printf("%02d/%02d/%04d", d.day, d.month, d.year);
}

void printHeader(const Person *p){
    printf("========================================\n");
    printf("Historia: %s\n", p->name);
    printf("Nascimento: "); printDate(p->birthdate); printf(" - %s\n", p->birthplace);
    printf("========================================\n\n");
}

void printFavorites(const Person *p){
    printf("Favoritos (amostral):\n");
    for(int i=0;i<p->favorites_count;i++){
        printf(" - %s\n", p->favorites[i]);
    }
    printf("\n");
}

void printIdols(const Person *p){
    printf("Ídolos e inspirações:\n");
    for(int i=0;i<p->idols_count;i++){
        printf(" - %s\n", p->idols[i]);
    }
    printf("\n");
}

void printTimeline(const Person *p){
    printf("Linha do tempo (eventos registrados):\n\n");
    for(int i=0;i<p->timeline_count;i++){
        Event e = p->timeline[i];
        printf("[%02d/%02d/%04d] %s\n", e.date.day, e.date.month, e.date.year, e.title);
        printf("   %s\n\n", e.description);
    }
}

int main(void){
    Person user = {0};

    // Identidade básica
    user.name = "Mickeias"; // nome associado à conta
    user.birthdate.day = 3; user.birthdate.month = 10; user.birthdate.year = 2001;
    user.birthplace = "Brasília, Brasil";

    // Favoritos (jogos, filmes, séries)
    user.favorites[0] = "GTA San Andreas";
    user.favorites[1] = "Elden Ring";
    user.favorites[2] = "Assassin's Creed: Black Flag / Valhalla";
    user.favorites[3] = "Harry Potter";
    user.favorites[4] = "Game of Thrones / House of the Dragon";
    user.favorites[5] = "Stranger Things";
    user.favorites[6] = "Breaking Bad";
    user.favorites_count = 7;

    // Ídolos e referências
    user.idols[0] = "Travis Scott";
    user.idols[1] = "Djavan";
    user.idols[2] = "Alan Turing";
    user.idols[3] = "George R.R. Martin";
    user.idols[4] = "Hideo Kojima";
    user.idols[5] = "Hidetaka Miyazaki";
    user.idols_count = 6;

    // Timeline (eventos importantes conhecidos)
    int tc = 0;

    user.timeline[tc++] = (Event){"Nascimento",
        "Nascimento em Brasília.", {3,10,2001}};

    user.timeline[tc++] = (Event){"Interesse por tecnologia e jogos",
        "Infância e adolescência marcadas por curiosidade por tecnologia; jogos como GTA e Assassin's Creed formaram gosto narrativo.", {0,0,0}};

    user.timeline[tc++] = (Event){"Fãs e shows de Travis Scott",
        "Grande fã de Travis Scott; foi a dois shows: Primavera Sound (SP) e Rock in Rio.", {0,0,0}};

    user.timeline[tc++] = (Event){"Entrada na faculdade de Ciência da Computação",
        "Começou o curso e iniciou projetos práticos e acadêmicos, vendo o código como forma de arte.", {0,0,0}};

    user.timeline[tc++] = (Event){"Amizade com Gustavo",
        "Fez amizade com Gustavo na faculdade — parceiro em projetos como ListBeat e ideias para equipe freelancer.", {0,0,0}};

    user.timeline[tc++] = (Event){"Projeto: ListBeat",
        "Aplicativo inspirado no Letterboxd para música: catálogo pessoal, avaliações, listas e histórico de músicas; desenvolvido em SwiftUI.", {0,0,0}};

    user.timeline[tc++] = (Event){"Projeto: Rolê (HackaTruck)",
        "Rede social para entretenimento cultural, entregue como projeto acadêmico.", {0,0,0}};

    user.timeline[tc++] = (Event){"Trabalhos e estudos em diversas tecnologias",
        "Experiência com Swift, SwiftUI, Java, modelagem de banco de dados, autenticação multifator e desenvolvimento web.", {0,0,0}};

    user.timeline_count = tc;

    // Impressão
    printHeader(&user);
    printFavorites(&user);
    printIdols(&user);
    printTimeline(&user);

    printf("FIM DA HISTORIA — até o momento presente (faculdade e projetos em andamento).\n");

    return 0;
}
